// When Stripe confirms host identity is verified
if (event.type === 'account.updated') {
  const acc = event.data.object
  if (acc.charges_enabled) {
    await supabase.from('profiles')
      .update({ stripe_account_id: acc.id, is_verified: true })
      .eq('stripe_account_id', acc.id)
  }
}

What happens when you publish
typescriptapi/listings/create.ts
// 1. Upload photos → Supabase Storage
const imageUrls = await uploadPhotos(body.photos, user.id)

// 2. Insert listing (hotel)
const { data: listing } = await supabase.from('listings').insert({
  host_id: user.id, type: 'hotel',
  title: body.title, images: imageUrls, status: 'live'
}).select().single()

// 3. Insert each SnapRoom room type
await supabase.from('snap_rooms').insert(
  body.rooms.map(r => ({
    listing_id: listing.id,
    tier: r.tier, name: r.name,
    price_per_night: r.price,
    units_available: r.units,
    bed_config: r.bed, view: r.view,
    amenities: r.amenities,
    max_guests: r.maxGuests
  }))
)

// 4. Email confirmation via Resend
await resend.emails.send({ to: user.email, ... })
